30 /**
31  * The Factory that creates an instance of a Flux client.
32  * 
33  * @author Jakub Bednar (bednar@github) (05/11/2018 10:10)
34  */
35 object InfluxDBClientScalaFactory {
36 
37 
38   /**
39    * Create an instance of the InfluxDB 2.0 client that is configured via `influx2.properties`.
40    * The `influx2.properties` has to be located on classpath.
41    *
42    * @return client
43    */
44   @Nonnull def create(): InfluxDBClientScala = {
45 
46     val options = InfluxDBClientOptions.builder()
47       .loadProperties()
48       .build()
49 
50     create(options)
51   }
52 
53   /**
54    * Create an instance of the InfluxDB 2.0 client. The url could be a connection string with various configurations.
55    *
56    * e.g.: "http://localhost:8086?readTimeout=5000&amp;connectTimeout=5000&amp;logLevel=BASIC
57    *
58    * @param connectionString connection string with various configurations.
59    * @return client
60    */
61   @Nonnull def create(connectionString: String): InfluxDBClientScala = {
62 
63     val options = InfluxDBClientOptions.builder()
64       .url(connectionString)
65       .build()
66 
67     create(options)
68   }
69 
70   /**
71    * Create an instance of the InfluxDB 2.0 reactive client.
72    *
73    * The ''username/password'' auth is based on
74    * [[http://bit.ly/http-basic-auth HTTP "Basic" authentication]]. The authorization expires when the
75    * [[http://bit.ly/session-lengthh time-to-live (TTL)]] (default 60 minutes) is reached
76    * and client produces [[com.influxdb.exceptions.UnauthorizedException]].
77    *
78    * @param url      the url to connect to the InfluxDB
79    * @param username the username to use in the basic auth
80    * @param password the password to use in the basic auth
81    * @return client
82    * @see [[InfluxDBClientOptions.Builder]]
83    */
84   @Nonnull def create(url: String, username: String, password: Array[Char]): InfluxDBClientScala = {
85 
86     val options = InfluxDBClientOptions.builder()
87       .url(url)
88       .authenticate(username, password)
89       .build()
90 
91     create(options)
92   }
93 
94   /**
95    * Create an instance of the InfluxDB 2.0 reactive client.
96    *
97    * @param url   the url to connect to the InfluxDB
98    * @param token the token to use for the authorization
99    * @return client
100    * @see [[InfluxDBClientOptions.Builder]]
101    */
102   @Nonnull def create(url: String, token: Array[Char]): InfluxDBClientScala = {
103 
104     create(url, token, null)
105   }
106 
107   /**
108    * Create an instance of the InfluxDB 2.0 reactive client.
109    *
110    * @param url   the url to connect to the InfluxDB
111    * @param token the token to use for the authorization
112    * @param org      the name of an organization
113    * @return client
114    * @see [[InfluxDBClientOptions.Builder]]
115    */
116   @Nonnull def create(url: String, token: Array[Char], org: String): InfluxDBClientScala = {
117 
118     create(url, token, org, null)
119   }
120 
121   /**
122    * Create an instance of the InfluxDB 2.0 reactive client.
123    *
124    * @param url   the url to connect to the InfluxDB
125    * @param token the token to use for the authorization
126    * @param org      the name of an organization
127    * @param bucket   the name of a bucket
128    * @return client
129    * @see [[InfluxDBClientOptions.Builder]]
130    */
131   @Nonnull def create(url: String, token: Array[Char], org: String, bucket: String): InfluxDBClientScala = {
132 
133     val options = InfluxDBClientOptions.builder()
134       .url(url)
135       .authenticateToken(token)
136       .org(org)
137       .bucket(bucket)
138       .build()
139 
140     create(options)
141   }
142 
143   /**
144    * Create an instance of the InfluxDB 2.0 reactive client.
145    *
146    * @param options the connection configuration
147    * @return client
148    */
149   @Nonnull def create(@Nonnull options: InfluxDBClientOptions): InfluxDBClientScala = {
150 
151     Arguments.checkNotNull(options, "InfluxDBClientOptions")
152 
153     new InfluxDBClientScalaImpl(options)
154   }
155 }
