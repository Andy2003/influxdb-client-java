38    * Get the Query client.
39    *
40    * @return the new client instance for the Query API

43 
44   /**
45    * Get the health of an instance.
46    *
47    * @return health of an instance
48    */

50

52    * Check the status of InfluxDB Server.
53    *
54    * @return true if server is healthy otherwise return false
55    */
56   override def ping: Boolean = ping(pingService.getPing)
57

59    * Returns the version of the connected InfluxDB Server.
60    *
61    * @return the version String, otherwise unknown.
62    */
63   override def version: String = version(pingService.getPing)
64

66    * Gets the [[LogLevel]] that is used for logging requests and responses.
67    *
68    * @return the [[LogLevel]] that is used for logging requests and responses

71 
72   /**
73    * Sets the log level for the request and response information.
74    *
75    * @param logLevel the log level to set.
76    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
77    */
78   override def setLogLevel(logLevel: LogLevel): InfluxDBClientScala = {
79     setLogLevel(loggingInterceptor, logLevel)
80     this
81   }
82 
83   /**
84    * Enable Gzip compress for http requests.
85    *
86    * Currently only the "Write" and "Query" endpoints supports the Gzip compression.
87    *
88    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
89    */
90   override def enableGzip(): InfluxDBClientScala = {
91     gzipInterceptor.enableGzip()
92     this
93   }
94 
95   /**
96    * Disable Gzip compress for http request body.
97    *
98    * @return the [[InfluxDBClientScala]] instance to be able to use it in a fluent manner.
99    */
100   override def disableGzip(): InfluxDBClientScala = {
101     gzipInterceptor.disableGzip()
102     this
103   }
104 
105   /**
106    * Returns whether Gzip compress for http request body is enabled.
107    *
108    * @return true if gzip is enabled.
109    */
110   override def isGzipEnabled: Boolean = gzipInterceptor.isEnabledGzip
